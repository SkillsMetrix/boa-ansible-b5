
from abc import ABC,abstractmethod

class RBI(ABC):
    @abstractmethod
    def deposit(self):
        pass
        
    def withdraw(self):
        print("withdraw")
    def checkBalance(self):
        print("checking Balance")


class SBI(RBI):
    def openFD(self):
        print("opening FD in SBI")
    def deposit(self):
        print("amount is deposited in SBI")


sbi= SBI()

sbi.deposit()
sbi.openFD()
