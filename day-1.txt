
class Student():
    def __init__(self,rollno,name,subject,marks):
        self.rollno=rollno
        self.name=name
        self.subject=subject
        self.marks=marks

    def displayData(self):

    def __str__(self):
        return f"The name and subject is {self.name} {self.subject}"
